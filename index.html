    ⚙ Apps Script 프록시 설정
    CORS 문제 해결을 위한 1회 설정 (5분 소요)
    왜 필요한가요?
    claude.ai 에서 구글 시트를 직접 불러오면 브라우저 보안 정책(CORS)에 막힙니다.
    구글 Apps Script를 중간 다리로 쓰면 이 제한을 우회할 수 있습니다.
    1 Apps Script 파일 만들기
    ① script.google.com 접속 → 새 프로젝트
    ② 아래 코드를 전체 선택 후 붙여넣기 (기존 내용 덮어쓰기) 
    function doGet(e) {
  const sheetId = e.parameter.sheetId;
  const tabName = e.parameter.tab;
  try {
    const ss = SpreadsheetApp.openById(sheetId);
    const sheet = ss.getSheetByName(tabName);
    if (!sheet) {
      return jsonResp({ error: "탭 없음: " + tabName });
    }
    const values = sheet.getDataRange().getValues();
    if (values.length < 2) return jsonResp({ data: [] });
    const headers = values[0].map(h => String(h).trim());
    const data = values.slice(1).map(row => {
      const obj = {};
      headers.forEach((h, i) => {
        const v = row[i];
        if (v instanceof Date) {
          obj[h] = Utilities.formatDate(v, "Asia/Seoul", "yyyy-MM-dd");
        } else {
          obj[h] = v === null || v === undefined ? "" : String(v);
        }
      });
      return obj;
    });
    return jsonResp({ data: data });
  } catch(err) {
    return jsonResp({ error: err.message });
  }
}
function jsonResp(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
